def conditional_probabilities(df):
    # (6) Price conditional probabilities
    price_diff = df['Adjusted'].diff()
    inc = (price_diff > 0).astype(int)
    dec = (price_diff < 0).astype(int)
    # Shift for t-1
    inc_prev = inc.shift(1)
    dec_prev = dec.shift(1)
    # Probabilities
    p_inc_inc = ((inc == 1) & (inc_prev == 1)).sum() / (inc_prev == 1).sum()
    p_inc_dec = ((inc == 1) & (dec_prev == 1)).sum() / (dec_prev == 1).sum()
    p_dec_inc = ((dec == 1) & (inc_prev == 1)).sum() / (inc_prev == 1).sum()
    p_dec_dec = ((dec == 1) & (dec_prev == 1)).sum() / (dec_prev == 1).sum()
    return p_inc_inc, p_inc_dec, p_dec_inc, p_dec_dec

def volume_conditional_probabilities(df):
    # (7) Volume conditional probabilities
    price_diff = df['Adjusted'].diff()
    volume_diff = df['Volume'].diff()
    price_inc = (price_diff > 0).astype(int)
    price_dec = (price_diff < 0).astype(int)
    vol_inc_prev = (volume_diff > 0).astype(int).shift(1)
    vol_dec_prev = (volume_diff < 0).astype(int).shift(1)
    # Probabilities
    p_inc_volinc = ((price_inc == 1) & (vol_inc_prev == 1)).sum() / (vol_inc_prev == 1).sum()
    p_inc_voldec = ((price_inc == 1) & (vol_dec_prev == 1)).sum() / (vol_dec_prev == 1).sum()
    p_dec_volinc = ((price_dec == 1) & (vol_inc_prev == 1)).sum() / (vol_inc_prev == 1).sum()
    p_dec_voldec = ((price_dec == 1) & (vol_dec_prev == 1)).sum() / (vol_dec_prev == 1).sum()
    return p_inc_volinc, p_inc_voldec, p_dec_volinc, p_dec_voldec

sigcount_slope = 0
alpha = 0.05

    st = np.random.normal(mu, sigma, T)  
    X = sm.add_constant(st)
    model = sm.OLS(returns.values, X)
    results = model.fit()
    
    if results.pvalues[0] < alpha:
        sigcount_intercept += 1
    if results.pvalues[1] < alpha:
        sigcount_slope += 1
